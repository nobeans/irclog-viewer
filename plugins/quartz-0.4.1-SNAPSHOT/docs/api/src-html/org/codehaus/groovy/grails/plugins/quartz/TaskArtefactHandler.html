016    package org.codehaus.groovy.grails.plugins.quartz;
017    
018    import org.codehaus.groovy.grails.commons.ArtefactHandlerAdapter;
019    import org.springframework.util.ReflectionUtils;
020    
021    import java.lang.reflect.Method;
022    
023    /**
024     * Grails artefact handler for task classes.
025     *
026     * @author Marc Palmer (marc@anyware.co.uk)
027     * @author Sergey Nebolsin (nebolsin@gmail.com)
028     * 
029     * @since 0.1
030     */
031    public class TaskArtefactHandler extends ArtefactHandlerAdapter {
032    
033        public static final String TYPE = "Task";
034    
035        public TaskArtefactHandler() {
036            super(TYPE, GrailsTaskClass.class, DefaultGrailsTaskClass.class, null);
037        }
038    
039        public boolean isArtefactClass(Class clazz) {
040            // class shouldn't be null and shoud ends with Job suffix
041            if(clazz == null || !clazz.getName().endsWith(DefaultGrailsTaskClass.JOB)) return false;
042            // and should have one of execute() or execute(JobExecutionContext) methods defined
043            Method method = ReflectionUtils.findMethod(clazz, GrailsTaskClassProperty.EXECUTE);
044            if(method == null) {
045                // we're using Object as a param here to allow groovy-style 'def execute(param)' method
046                method = ReflectionUtils.findMethod(clazz, GrailsTaskClassProperty.EXECUTE, new Class[]{Object.class});
047            }
048            return method != null;
049        }
050    }
